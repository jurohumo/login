comando para inicializar un proyecto en git
-- git init

comando para agregar el seguimiento de un archivo
--git add <file>

comando para agregar todos los documentos cambiados
-- git add .

comando para la generacion de un punto de restauracion
-- git commit
-- git commit -m "<mensaje>"

comando para ver las modificaciones en el proyecto
-- git status

comando para ver los cambios realizados en el proyecto
-- git log

comando para descartar los cambios realizados en un archivo
-- git checkout -- <file>

comando para ver lo que se ha modificado en un archivo
-- git diff <file>

comando para verificar en que branch estas
--git branch

comando para generar un nuevo branch
-- git branch <nombre>

comando para cambiar de branch
-- git checkout <nombre>
